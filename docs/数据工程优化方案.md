### 2.1 构建铜-银-金三层数据架构

```python
# core/data_quality/pipeline_manager.py
"""
数据管道管理层 - 实现铜银金架构
"""

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Optional, Dict, Any, List
from delta.tables import DeltaTable
from pyspark.sql import SparkSession
from pyspark.sql.functions import col, current_timestamp, sha2, concat_ws, lit

class DataTier(Enum):
    """数据层级别"""
    BRONZE = "bronze"      # 原始层：仅追加、零转换
    SILVER = "silver"      # 清洗层：去重、标准化
    GOLD = "gold"          # 消费层：业务聚合、SLA保障

@dataclass
class DataContract:
    """数据契约定义"""
    collection: str
    tier: DataTier
    primary_key: List[str]
    partition_cols: List[str]
    null_policy: Dict[str, str]  # 字段 -> 策略(drop/fill/default/reject)
    sla_minutes: int = 60
    quality_threshold: float = 0.95

class DataPipelineManager:
    """
    统一数据管道管理器
    
    职责：
    1. 管理铜银金三层数据流动
    2. 执行数据质量验证
    3. 维护数据血缘
    """
    
    QUALITY_COLUMNS = {
        "case_library_v2": {
            "required": ["case_id", "content", "scene_type", "quality_score"],
            "null_tolerance": {"scene_type": 0.0, "quality_score": 0.0}
        },
        "writing_techniques_v2": {
            "required": ["technique_id", "name", "dimension", "content"],
            "null_tolerance": {}
        }
    }
    
    def __init__(self, storage_path: str):
        self.storage_path = storage_path
        self._spark = None
        
    @property
    def spark(self) -> SparkSession:
        if self._spark is None:
            self._spark = SparkSession.builder \
                .config("spark.sql.extensions", "io.delta.sql.DeltaSparkSessionExtension") \
                .config("spark.sql.catalog.spark_catalog", "org.apache.spark.sql.delta.catalog.DeltaCatalog") \
                .getOrCreate()
        return self._spark
    
    def ingest_bronze(self, source_path: str, collection: str) -> Dict[str, Any]:
        """
        铜层摄入：原始数据，仅追加，零转换
        
        Args:
            source_path: 源文件路径
            collection: 目标Collection名称
            
        Returns:
            摄入统计信息
        """
        bronze_path = f"{self.storage_path}/bronze/{collection}"
        
        # 读取源数据
        df = self.spark.read.format("json").option("inferSchema", "true").load(source_path)
        
        # 添加元数据（不修改原始数据）
        df = df.withColumn("_ingested_at", current_timestamp()) \
               .withColumn("_source_system", lit(collection)) \
               .withColumn("_source_file", col("_metadata.file_path")) \
               .withColumn("_ingestion_hash", sha2(concat_ws("||", *[col(c) for c in df.columns]), 256))
        
        # 仅追加模式
        df.write.format("delta") \
            .mode("append") \
            .option("mergeSchema", "true") \
            .save(bronze_path)
        
        return {
            "tier": "bronze",
            "collection": collection,
            "records_ingested": df.count(),
            "partition": datetime.now().strftime("%Y-%m-%d")
        }
    
    def upsert_silver(self, bronze_table: str, silver_table: str, 
                      pk_cols: List[str], collection: str) -> None:
        """
        银层清洗：去重、标准化、null处理
        
        Args:
            bronze_table: 铜层表路径
            silver_table: 银层表路径
            pk_cols: 主键列
            collection: Collection名称
        """
        source = self.spark.read.format("delta").load(bronze_table)
        
        # 获取契约定义
        contract = self.QUALITY_COLUMNS.get(collection, {})
        
        # 1. 质量过滤：去除不符合要求的记录
        required_cols = contract.get("required", [])
        for req_col in required_cols:
            if req_col in source.columns:
                source = source.filter(col(req_col).isNotNull())
        
        # 2. 去重：根据主键保留最新记录
        from pyspark.sql.window import Window
        from pyspark.sql.functions import row_number, desc
        
        w = Window.partitionBy(*pk_cols).orderBy(desc("_ingested_at"))
        deduped = source.withColumn("_rank", row_number().over(w)) \
                        .filter(col("_rank") == 1) \
                        .drop("_rank")
        
        # 3. 添加质量分数
        deduped = self._compute_row_quality(deduped, collection)
        
        # 4. 审计列
        deduped = deduped.withColumn("_silver_processed_at", current_timestamp()) \
                         .withColumn("_silver_contract_version", lit("1.0"))
        
        # 5. UPSERT到银层
        if DeltaTable.isDeltaTable(self.spark, silver_table):
            target = DeltaTable.forPath(self.spark, silver_table)
            merge_condition = " AND ".join([f"target.{c} = source.{c}" for c in pk_cols])
            
            target.alias("target").merge(deduped.alias("source"), merge_condition) \
                .whenMatchedUpdateAll() \
                .whenNotMatchedInsertAll() \
                .execute()
        else:
            deduped.write.format("delta").mode("overwrite").save(silver_table)
    
    def _compute_row_quality(self, df, collection: str) -> "DataFrame":
        """计算行级质量分数"""
        from pyspark.sql.functions import when, col
        
        contract = self.QUALITY_COLUMNS.get(collection, {})
        required = contract.get("required", [])
        
        # 基础分：是否有所有必需字段
        required_count = len([c for c in required if c in df.columns])
        base_score = when(required_count == len(required), 1.0) \
                     .otherwise(required_count / max(len(required), 1))
        
        # 返回质量分数列
        return df.withColumn("_row_quality_score", base_score)
    
    def build_gold(self, silver_table: str, gold_table: str,
                   aggregation_sql: str, partition_col: str) -> None:
        """
        金层聚合：业务指标，查询优化
        
        Args:
            silver_table: 银层表路径
            gold_table: 金层表路径
            aggregation_sql: 聚合SQL或函数
            partition_col: 分区列
        """
        silver = self.spark.read.format("delta").load(silver_table)
        
        # 执行聚合（根据具体业务逻辑）
        gold = self._aggregate_for_consumption(silver, aggregation_sql)
        
        # 添加刷新时间戳和SLA元数据
        gold = gold.withColumn("_refreshed_at", current_timestamp()) \
                   .withColumn("_gold_partition", col(partition_col))
        
        # 使用replaceWhere优化查询
        min_partition = gold.agg({partition_col: "min"}).collect()[0][0]
        max_partition = gold.agg({partition_col: "max"}).collect()[0][0]
        
        gold.write.format("delta") \
            .mode("overwrite") \
            .option("replaceWhere", f"{partition_col} >= '{min_partition}' AND {partition_col} <= '{max_partition}'") \
            .partitionBy("_gold_partition") \
            .save(gold_table)
    
    def _aggregate_for_consumption(self, df, aggregation_sql: str):
        """根据业务逻辑执行聚合"""
        # 实际实现根据具体业务需求定制
        return df
    
    def validate_tier(self, table_path: str, tier: DataTier) -> Dict[str, Any]:
        """
        验证数据层质量
        
        Returns:
            验证结果包含：record_count, null_rates, quality_pass_rate
        """
        df = self.spark.read.format("delta").load(table_path)
        
        total = df.count()
        
        # 计算各列null率
        null_rates = {}
        for col_name in df.columns:
            if col_name.startswith("_"):
                continue  # 跳过元数据列
            null_count = df.filter(col(col_name).isNull()).count()
            null_rates[col_name] = null_count / total if total > 0 else 0
        
        # 计算质量分数统计
        if "_row_quality_score" in df.columns:
            quality_stats = df.select(
                col("_row_quality_score").mean().alias("avg_quality"),
                col("_row_quality_score").min().alias("min_quality"),
                col("_row_quality_score").max().alias("max_quality")
            ).collect()[0]
        else:
            quality_stats = {"avg_quality": 1.0, "min_quality": 1.0, "max_quality": 1.0}
        
        return {
            "tier": tier.value,
            "record_count": total,
            "null_rates": null_rates,
            "quality_stats": quality_stats,
            "validated_at": datetime.now().isoformat()
        }
```

```python
"""
自适应质量阈值系统

根据历史数据分布自动调整质量阈值，而非使用固定值
"""

from dataclasses import dataclass
from typing import Dict, List, Optional, Tuple
import numpy as np
from scipy import stats

@dataclass
class QualityThreshold:
    """质量阈值配置"""
    metric: str
    lower_bound: float
    upper_bound: float
    method: str  # "percentile", "zscore", "iqr"
    confidence_level: float = 0.95

class AdaptiveThresholdManager:
    """
    自适应阈值管理器
    
    根据历史数据统计特征自动计算合理的质量阈值
    """
    
    def __init__(self, history_window: int = 1000):
        self.history_window = history_window
        self._history: Dict[str, List[float]] = {}
        self._thresholds: Dict[str, QualityThreshold] = {}
    
    def record_metric(self, metric: str, value: float) -> None:
        """记录质量指标值"""
        if metric not in self._history:
            self._history[metric] = []
        self._history[metric].append(value)
        
        # 保持滑动窗口大小
        if len(self._history[metric]) > self.history_window:
            self._history[metric] = self._history[metric][-self.history_window:]
    
    def compute_threshold(self, metric: str, 
                         method: str = "percentile",
                         confidence_level: float = 0.95) -> QualityThreshold:
        """
        计算自适应阈值
        
        Args:
            metric: 指标名称
            method: 计算方法
                - "percentile": 百分位数法（推荐）
                - "zscore": Z分数法
                - "iqr": 四分位距法
            confidence_level: 置信水平
        """
        if metric not in self._history or len(self._history[metric]) < 30:
            # 数据不足，返回保守默认值
            return QualityThreshold(
                metric=metric,
                lower_bound=0.0,
                upper_bound=1.0,
                method=method,
                confidence_level=confidence_level
            )
        
        values = np.array(self._history[metric])
        
        if method == "percentile":
            # 百分位数法：使用1.5倍IQR边界
            q1 = np.percentile(values, 25)
            q3 = np.percentile(values, 75)
            iqr = q3 - q1
            lower = q1 - 1.5 * iqr
            upper = q3 + 1.5 * iqr
            
        elif method == "zscore":
            # Z分数法：均值±2.5倍标准差
            mean = np.mean(values)
            std = np.std(values)
            lower = mean - 2.5 * std
            upper = mean + 2.5 * std
            
        elif method == "iqr":
            # IQR法：更严格的边界
            q1 = np.percentile(values, 10)
            q3 = np.percentile(values, 90)
            iqr = q3 - q1
            lower = q1 - 1.0 * iqr
            upper = q3 + 1.0 * iqr
        
        # 限制在有效范围内
        lower = max(0.0, lower)
        upper = min(1.0, upper)
        
        threshold = QualityThreshold(
            metric=metric,
            lower_bound=lower,
            upper_bound=upper,
            method=method,
            confidence_level=confidence_level
        )
        
        self._thresholds[metric] = threshold
        return threshold
    
    def is_within_threshold(self, metric: str, value: float) -> Tuple[bool, Optional[str]]:
        """
        检查值是否在阈值范围内
        
        Returns:
            (是否合规, 违规原因)
        """
        if metric not in self._thresholds:
            # 未定义阈值，默认合规
            return True, None
        
        threshold = self._thresholds[metric]
        
        if value < threshold.lower_bound:
            return False, f"低于下限 {threshold.lower_bound:.3f}"
        if value > threshold.upper_bound:
            return False, f"超过上限 {threshold.upper_bound:.3f}"
        
        return True, None
    
    def get_thresholds_report(self) -> Dict:
        """生成阈值报告"""
        return {
            metric: {
                "lower_bound": t.lower_bound,
                "upper_bound": t.upper_bound,
                "method": t.method,
                "sample_size": len(self._history.get(metric, []))
            }
            for metric, t in self._thresholds.items()
        }
```

```python
        texts,
        return_dense=True,
        return_sparse=False,
        return_colbert_vecs=False

        # L2归一化
        norms = np.linalg.norm(vectors, axis=1, keepdims=True)

    批量嵌入文本
    
    支持向量缓存和智能批处理

    start = time.time()
    
    # 1. 检查缓存
    cache_keys = [_get_cache_key(t, request.pooling, request.normalize) for t in request.texts]
    cached_results = {}
    uncached_indices = []
    uncached_texts = []
    
    for i, key in enumerate(cache_keys):
        if key in _cache:
            cached_results[i] = _cache[key]
        else:
            uncached_indices.append(i)
            uncached_texts.append(request.texts[i])
    
    # 2. 批量编码未缓存文本
    if uncached_texts:
        loop = asyncio.get_event_loop()
        new_vectors = await loop.run_in_executor(
            _executor,
            _encode_texts,
            uncached_texts,
            request.pooling,
            request.normalize
        )
        
        # 3. 更新缓存
        for i, vec in zip(uncached_indices, new_vectors):
            key = cache_keys[i]
            if len(_cache) < _cache_max_size:
                _cache[key] = vec
            cached_results[i] = vec
    
    # 4. 组装结果
    vectors = [cached_results[i] for i in range(len(request.texts))]
    
    latency_ms = (time.time() - start) * 1000
    
    return EmbedResponse(
        vectors=vectors,
        model=request.model,
        cached=len(cached_results) - len(uncached_texts),
        latency_ms=latency_ms
    )
```

### 3.2 Qdrant连接池和重试机制

```python
# core/database/qdrant_pool.py
"""
Qdrant连接池管理

提供：
1. 连接池复用
2. 自动重试机制
3. 熔断器模式
4. 指标监控
"""

import time
import threading
from typing import Optional, Callable, Any
from dataclasses import dataclass
from enum import Enum
from functools import wraps

from qdrant_client import QdrantClient
from qdrant_client.http.exceptions import UnexpectedResponse, RpcError

class CircuitState(Enum):
    """熔断器状态"""
    CLOSED = "closed"      # 正常
    OPEN = "open"          # 熔断
    HALF_OPEN = "half_open"  # 半开

@dataclass
class PoolConfig:
    """连接池配置"""
    max_connections: int = 10
    max_retries: int = 3
    retry_base_delay: float = 1.0
    retry_max_delay: float = 30.0
    circuit_threshold: int = 5       # 熔断阈值（失败次数）
    circuit_reset_timeout: float = 60.0  # 熔断重置时间
    timeout: float = 30.0

class CircuitBreaker:
    """熔断器实现"""
    
    def __init__(self, threshold: int = 5, reset_timeout: float = 60.0):
        self.threshold = threshold
        self.reset_timeout = reset_timeout
        self._state = CircuitState.CLOSED
        self._failure_count = 0
        self._last_failure_time: Optional[float] = None
        self._lock = threading.Lock()
    
    @property
    def state(self) -> CircuitState:
        with self._lock:
            if self._state == CircuitState.OPEN:
                # 检查是否应该转换到半开
                if (time.time() - self._last_failure_time) > self.reset_timeout:
                    self._state = CircuitState.HALF_OPEN
            return self._state
    
    def record_success(self) -> None:
        with self._lock:
            self._failure_count = 0
            self._state = CircuitState.CLOSED
    
    def record_failure(self) -> None:
        with self._lock:
            self._failure_count += 1
            self._last_failure_time = time.time()
            if self._failure_count >= self.threshold:
                self._state = CircuitState.OPEN


class QdrantConnectionPool:
    """
    Qdrant连接池
    
    提供连接复用、自动重试、熔断保护
    """
    
    _instance: Optional["QdrantConnectionPool"] = None
    _lock = threading.Lock()
    
    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._initialized = False
        return cls._instance
    
    def __init__(self, config: Optional[PoolConfig] = None):
        if self._initialized:
            return
        
        self.config = config or PoolConfig()
        self._client: Optional[QdrantClient] = None
        self._client_lock = threading.Lock()
        self._circuit_breaker = CircuitBreaker(
            threshold=self.config.circuit_threshold,
            reset_timeout=self.config.circuit_reset_timeout
        )
        self._stats = {
            "total_requests": 0,
            "successful_requests": 0,
            "failed_requests": 0,
            "retried_requests": 0,
            "circuit_opens": 0
        }
        self._stats_lock = threading.Lock()
        self._initialized = True
    
    @property
    def client(self) -> QdrantClient:
        """获取Qdrant客户端（懒加载）"""
        if self._client is None:
            with self._client_lock:
                if self._client is None:
                    self._client = QdrantClient(
                        host="localhost",
                        port=6333,
                        timeout=self.config.timeout
                    )
        return self._client
    
    def _exponential_backoff(self, attempt: int) -> float:
        """指数退避延迟"""
        delay = min(
            self.config.retry_base_delay * (2 ** attempt),
            self.config.retry_max_delay
        )
        # 添加随机抖动
        import random
        return delay * (0.5 + random.random())
    
    def execute_with_retry(
        self,
        operation: Callable[[], Any],
        operation_name: str = "operation"
    ) -> Any:
        """
        带重试机制的执行
        
        Args:
            operation: 要执行的操作
            operation_name: 操作名称（用于日志）
            
        Returns:
            操作结果
        """
        # 检查熔断器
        if self._circuit_breaker.state == CircuitState.OPEN:
            raise CircuitBreakerOpenError(
                f"Circuit breaker is OPEN for {operation_name}"
            )
        
        last_error = None
        
        for attempt in range(self.config.max_retries):
            try:
                with self._stats_lock:
                    self._stats["total_requests"] += 1
                
                result = operation()
                
                # 成功
                self._circuit_breaker.record_success()
                with self._stats_lock:
                    self._stats["successful_requests"] += 1
                
                if attempt > 0:
                    with self._stats_lock:
                        self._stats["retried_requests"] += 1
                
                return result
                
            except (UnexpectedResponse, RpcError, ConnectionError) as e:
                last_error = e
                self._circuit_breaker.record_failure()
                
                with self._stats_lock:
                    self._stats["failed_requests"] += 1
                
                if attempt < self.config.max_retries - 1:
                    delay = self._exponential_backoff(attempt)
                    time.sleep(delay)
                else:
                    # 熔断
                    with self._stats_lock:
                        self._stats["circuit_opens"] += 1
        
        raise last_error
    
    def get_stats(self) -> dict:
        """获取连接池统计"""
        with self._stats_lock:
            stats = self._stats.copy()
        
        stats["circuit_state"] = self._circuit_breaker.state.value
        stats["success_rate"] = (
            stats["successful_requests"] / max(stats["total_requests"], 1)
        )
        
        return stats
    
    def reset(self) -> None:
        """重置连接池"""
        with self._client_lock:
            if self._client:
                self._client.close()
            self._client = None
        
        with self._stats_lock:
            self._stats = {
                "total_requests": 0,
                "successful_requests": 0,
                "failed_requests": 0,
                "retried_requests": 0,
                "circuit_opens": 0
            }


# 便捷装饰器
def with_connection_pool(operation_name: str = "qdrant_operation"):
    """
    装饰器：为Qdrant操作添加连接池和重试逻辑
    
    Usage:
        @with_connection_pool("search_cases")
        def search_cases(query: str, limit: int):
            return pool.client.query_points(...)
    """
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs):
            pool = QdrantConnectionPool()
            
            def _operation():
                return func(*args, **kwargs)
            
            return pool.execute_with_retry(_operation, operation_name)
        
        return wrapper
    return decorator


class CircuitBreakerOpenError(Exception):
    """熔断器打开异常"""
    pass
```

### 3.3 增量管道优化

```python
# core/data_pipeline/incremental_processor.py
"""
增量数据处理管道

优化策略：
1. CDC变更检测
2. 断点续传
3. 批处理优化
"""

from dataclasses import dataclass, field
from datetime import datetime
from typing import Dict, List, Optional, Any, Set, Callable
from pathlib import Path
import json
import hashlib

@dataclass
class Checkpoint:
    """处理检查点"""
    collection: str
    last_processed_id: str
    last_processed_time: str
    processed_count: int
    checksum: str  # 数据校验和
    
@dataclass
class ProcessingBatch:
    """处理批次"""
    batch_id: str
    items: List[Dict]
    batch_size: int
    checksum: str
    created_at: str = field(default_factory=lambda: datetime.now().isoformat())

class IncrementalPipeline:
    """
    增量数据处理管道
    
    特性：
    1. 幂等处理：相同输入产生相同输出
    2. 断点续传：失败后可从检查点恢复
    3. 变更检测：只处理新增/变更数据
    4. 批量优化：智能批大小调整
    """
    
    def __init__(self, checkpoint_dir: str, collection: str):
        self.checkpoint_dir = Path(checkpoint_dir)
        self.checkpoint_dir.mkdir(parents=True, exist_ok=True)
        self.collection = collection
        self._checkpoint: Optional[Checkpoint] = self._load_checkpoint()
    
    def _load_checkpoint(self) -> Optional[Checkpoint]:
        """加载检查点"""
        checkpoint_file = self.checkpoint_dir / f"{self.collection}_checkpoint.json"
        if checkpoint_file.exists():
            with open(checkpoint_file, "r", encoding="utf-8") as f:
                data = json.load(f)
                return Checkpoint(**data)
        return None
    
    def _save_checkpoint(self, checkpoint: Checkpoint) -> None:
        """保存检查点"""
        checkpoint_file = self.checkpoint_dir / f"{self.collection}_checkpoint.json"
        with open(checkpoint_file, "w", encoding="utf-8") as f:
            json.dump({
                "collection": checkpoint.collection,
                "last_processed_id": checkpoint.last_processed_id,
                "last_processed_time": checkpoint.last_processed_time,
                "processed_count": checkpoint.processed_count,
                "checksum": checkpoint.checksum
            }, f, ensure_ascii=False, indent=2)
    
    def compute_data_checksum(self, data: List[Dict]) -> str:
        """计算数据校验和"""
        # 使用内容哈希
        content = json.dumps(data, sort_keys=True, ensure_ascii=False)
        return hashlib.sha256(content.encode()).hexdigest()[:16]
    
    def detect_changes(
        self,
        current_data: List[Dict],
        id_field: str = "id",
        compare_fields: Optional[List[str]] = None
    ) -> List[Dict]:
        """
        检测数据变更
        
        Args:
            current_data: 当前数据
            id_field: ID字段名
            compare_fields: 需要比较的字段列表（None表示所有字段）
            
        Returns:
            只包含新增或变更的记录
        """
        if self._checkpoint is None:
            # 无检查点，全量处理
            return current_data
        
        # 加载上次处理的数据快照（简化实现，实际应使用数据库）
        previous_snapshot = self._load_previous_snapshot()
        
        changes = []
        for item in current_data:
            item_id = item.get(id_field)
            if item_id is None:
                continue
            
            if item_id not in previous_snapshot:
                # 新增
                changes.append(item)
            else:
                # 检查变更
                prev_item = previous_snapshot[item_id]
                if self._has_changed(item, prev_item, compare_fields):
                    changes.append(item)
        
        return changes
    
    def _load_previous_snapshot(self) -> Dict[str, Dict]:
        """加载上次快照"""
        snapshot_file = self.checkpoint_dir / f"{self.collection}_snapshot.json"
        if snapshot_file.exists():
            with open(snapshot_file, "r", encoding="utf-8") as f:
                return json.load(f)
        return {}
    
    def _save_snapshot(self, data: List[Dict], id_field: str) -> None:
        """保存数据快照"""
        snapshot_file = self.checkpoint_dir / f"{self.collection}_snapshot.json"
        snapshot = {item.get(id_field): item for item in data if item.get(id_field)}
        with open(snapshot_file, "w", encoding="utf-8") as f:
            json.dump(snapshot, f, ensure_ascii=False)
    
    def _has_changed(
        self,
        current: Dict,
        previous: Dict,
        compare_fields: Optional[List[str]]
    ) -> bool:
        """检查是否有变更"""
        if compare_fields is None:
            compare_fields = list(set(current.keys()) | set(previous.keys()))
        
        for field in compare_fields:
            if current.get(field) != previous.get(field):
                return True
        return False
    
    def create_batch(
        self,
        items: List[Dict],
        batch_size: int = 100
    ) -> List[ProcessingBatch]:
        """创建处理批次"""
        batches = []
        
        for i in range(0, len(items), batch_size):
            batch_items = items[i:i + batch_size]
            batch = ProcessingBatch(
                batch_id=f"{self.collection}_{i // batch_size}",
                items=batch_items,
                batch_size=len(batch_items),
                checksum=self.compute_data_checksum(batch_items)
            )
            batches.append(batch)
        
        return batches
    
    def process_incremental(
        self,
        data_source: Callable[[Optional[str]], List[Dict]],
        processor: Callable[[List[Dict]], int],  # 返回处理数量
        id_field: str = "id",
        batch_size: int = 100,
        checkpoint_interval: int = 1000
    ) -> Dict[str, Any]:
        """
        执行增量处理
        
        Args:
            data_source: 数据源函数，接收上次处理ID返回数据
            processor: 处理函数，接收批次数据返回处理数量
            id_field: ID字段名
            batch_size: 批大小
            checkpoint_interval: 检查点保存间隔
            
        Returns:
            处理统计
        """
        start_time = datetime.now()
        
        # 确定起始位置
        start_id = self._checkpoint.last_processed_id if self._checkpoint else None
        
        # 获取数据
        all_data = data_source(start_id)
        
        # 检测变更
        changed_data = self.detect_changes(all_data, id_field)
        
        if not changed_data:
            return {
                "status": "no_changes",
                "processed": 0,
                "duration_seconds": 0
            }
        
        # 创建批次
        batches = self.create_batch(changed_data, batch_size)
        
        # 处理批次
        total_processed = 0
        last_item = None
        
        for batch in batches:
            processed = processor(batch.items)
            total_processed += processed
            
            # 更新检查点
            if batch.items:
                last_item = batch.items[-1]
            
            if total_processed % checkpoint_interval == 0:
                self._save_checkpoint(Checkpoint(
                    collection=self.collection,
                    last_processed_id=str(last_item.get(id_field)) if last_item else "",
                    last_processed_time=datetime.now().isoformat(),
                    processed_count=total_processed,
                    checksum=batch.checksum
                ))
        
        # 最终检查点
        if last_item:
            self._save_checkpoint(Checkpoint(
                collection=self.collection,
                last_processed_id=str(last_item.get(id_field)),
                last_processed_time=datetime.now().isoformat(),
                processed_count=total_processed,
                checksum=self.compute_data_checksum(changed_data)
            ))
            
            # 更新快照
            self._save_snapshot(changed_data, id_field)
        
        duration = (datetime.now() - start_time).total_seconds()
        
        return {
            "status": "completed",
            "collection": self.collection,
            "total_processed": total_processed,
            "new_or_changed": len(changed_data),
            "batches": len(batches),
            "duration_seconds": duration,
            "last_processed_id": last_item.get(id_field) if last_item else None
        }
```

```python
# core/monitoring/retrieval_monitor.py
"""
检索质量监控系统

指标：
1. Recall@K - 检索召回率
2. NDCG@K - 归一化折损累积增益

4. 结果多样性

from datetime import datetime
from typing import List, Dict, Optional, Tuple
from enum import Enum
import numpy as np

class RelevanceLevel(Enum):

    IRRELEVANT = 0  # 不相关

class RetrievalResult:

class RetrievalMonitor:
    """
    检索质量监控器
    
    功能：
    1. 记录检索日志
    2. 计算质量指标
    3. 生成监控报告
    """

        self._results: List[RetrievalResult] = []
        self._ground_truth: Dict[str, List[str]] = {}  # query_id -> 真正相关ID列表
    
    def set_ground_truth(self, query_id: str, relevant_ids: List[str]) -> None:

        query_text: str,
        returned_ids: List[str],
        retrieval_time_ms: float
    ) -> None:
        """记录单次检索"""
        # 获取相关性标签（如果有）
        relevance_labels = []
        if query_id in self._ground_truth:
            relevant_set = set(self._ground_truth[query_id])
            for doc_id in returned_ids:
                if doc_id in relevant_set:
                    relevance_labels.append(RelevanceLevel.RELEVANT)
                else:
                    relevance_labels.append(RelevanceLevel.IRRELEVANT)
        
        result = RetrievalResult(
            query_id=query_id,
            query_text=query_text,
            returned_ids=returned_ids,
            relevance_labels=relevance_labels,
            retrieval_time_ms=retrieval_time_ms
        )
        
        self._results.append(result)
    
    def compute_recall_at_k(self, k: int) -> float:
        """计算Recall@K"""
        if not self._results or not self._ground_truth:
            return 0.0
        
        recalls = []
        for result in self._results:
            if result.query_id not in self._ground_truth:
                continue
            
            relevant = set(self._ground_truth[result.query_id])
            retrieved_k = set(result.returned_ids[:k])
            
            if len(relevant) > 0:
                recall = len(retrieved_k & relevant) / len(relevant)
                recalls.append(recall)
        
        return np.mean(recalls) if recalls else 0.0
    
    def compute_ndcg_at_k(self, k: int) -> float:
        """计算NDCG@K"""
        if not self._results or not self._ground_truth:
            return 0.0
        
        ndcgs = []
        for result in self._results:
            if result.query_id not in self._ground_truth:
                continue
            
            # DCG@k
            dcg = 0.0
            for i, doc_id in enumerate(result.returned_ids[:k]):
                rel_label = result.relevance_labels[i] if i < len(result.relevance_labels) else RelevanceLevel.IRRELEVANT
                dcg += (2 ** rel_label.value - 1) / np.log2(i + 2)
            
            # IDCG@k（理想情况下的DCG）
            relevant_count = len(self._ground_truth[result.query_id])
            idcg = sum((2 ** RelevanceLevel.RELEVANT.value - 1) / np.log2(i + 2) 
                      for i in range(min(k, relevant_count)))
            
            # NDCG@k
            ndcg = dcg / idcg if idcg > 0 else 0.0
            ndcgs.append(ndcg)
        
        return np.mean(ndcgs) if ndcgs else 0.0
    
    def compute_average_precision(self) -> float:
        """计算平均精度(MAP)"""
        if not self._results or not self._ground_truth:
            return 0.0
        
        aps = []
        for result in self._results:
            if result.query_id not in self._ground_truth:
                continue
            
            relevant = set(self._ground_truth[result.query_id])
            
            precisions = []
            relevant_found = 0
            for i, doc_id in enumerate(result.returned_ids):
                if doc_id in relevant:
                    relevant_found += 1
                    precision_at_i = relevant_found / (i + 1)
                    precisions.append(precision_at_i)
            
            ap = np.mean(precisions) if precisions else 0.0
            aps.append(ap)
        
        return np.mean(aps) if aps else 0.0
    
    def compute_latency_percentiles(self) -> Tuple[float, float, float]:
        """计算延迟百分位数"""
        latencies = [r.retrieval_time_ms for r in self._results]
        
        if not latencies:
            return 0.0, 0.0, 0.0
        
        return (
            np.percentile(latencies, 50),
            np.percentile(latencies, 95),
            np.percentile(latencies, 99)
        )
    
    def compute_diversity(self, epsilon: float = 0.5) -> float:
        """
        计算结果多样性（MMR风格）
        
        Args:
            epsilon: 多样性权重（0=只管相关性，1=只管多样性）
        """
        if not self._results:
            return 0.0
        
        diversities = []
        for result in self._results:
            if len(result.returned_ids) < 2:
                diversities.append(1.0)
                continue
            
            # 简化的多样性计算：基于ID的Jaccard距离
            # 实际应基于内容的语义相似度
            total_distance = 0
            for i in range(len(result.returned_ids)):
                for j in range(i + 1, len(result.returned_ids)):
                    id_i = result.returned_ids[i]
                    id_j = result.returned_ids[j]
                    # 假设不同ID之间有相似性
                    total_distance += 1.0 if id_i != id_j else 0.0
            
            avg_distance = total_distance / (len(result.returned_ids) * (len(result.returned_ids) - 1) / 2)
            diversities.append(avg_distance)
        
        return np.mean(diversities)
    
    def get_metrics(self) -> RetrievalMetrics:
        """获取完整指标"""
        p50, p95, p99 = self.compute_latency_percentiles()
        
        return RetrievalMetrics(
            recall_at_k=self.compute_recall_at_k(max(self.k_values)),
            ndcg_at_k=self.compute_ndcg_at_k(max(self.k_values)),
            average_precision=self.compute_average_precision(),
            latency_p50_ms=p50,
            latency_p95_ms=p95,
            latency_p99_ms=p99,
            result_diversity=self.compute_diversity()
        )
    
    def generate_report(self) -> Dict:
        """生成监控报告"""
        metrics = self.get_metrics()
        
        report = {
            "generated_at": datetime.now().isoformat(),
            "total_queries": len(self._results),
            "metrics": {
                f"recall@{max(self.k_values)}": f"{metrics.recall_at_k:.3f}",
                f"ndcg@{max(self.k_values)}": f"{metrics.ndcg_at_k:.3f}",
                "map": f"{metrics.average_precision:.3f}",
                "latency_p50_ms": f"{metrics.latency_p50_ms:.1f}",
                "latency_p95_ms": f"{metrics.latency_p95_ms:.1f}",
                "latency_p99_ms": f"{metrics.latency_p99_ms:.1f}",
                "result_diversity": f"{metrics.result_diversity:.3f}"
            }
        }
        
        # 添加各K值的Recall
        for k in self.k_values:
            report["metrics"][f"recall@{k}"] = f"{self.compute_recall_at_k(k):.3f}"
            report["metrics"][f"ndcg@{k}"] = f"{self.compute_ndcg_at_k(k):.3f}"
        
        return report
```

```python
# core/monitoring/observability_dashboard.py
"""
可观测性仪表板

关键监控指标：
1. 数据新鲜度（各Collection最后更新时间）
2. 管道SLA达成率
3. 数据质量趋势
4. 系统健康状态
"""

from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import Dict, List, Optional
from enum import Enum

class HealthStatus(Enum):
    HEALTHY = "healthy"
    DEGRADED = "degraded"
    DOWN = "down"

@dataclass
class SlaDefinition:
    """SLA定义"""
    name: str
    target_freshness_minutes: int  # 目标新鲜度（分钟）
    warning_threshold_minutes: int  # 警告阈值
    critical_threshold_minutes: int  # 严重阈值

@dataclass
class PipelineHealth:
    """管道健康状态"""
    pipeline_name: str
    status: HealthStatus
    last_run: Optional[str]
    next_run: Optional[str]
    records_processed: int
    failed_records: int
    error_rate: float

class ObservabilityDashboard:
    """
    可观测性仪表板
    
    提供系统运行状态的完整视图
    """
    
    # 默认SLA定义
    DEFAULT_SLAS = {
        "case_library_v2": SlaDefinition(
            name="案例库新鲜度",
            target_freshness_minutes=60,
            warning_threshold_minutes=120,
            critical_threshold_minutes=240
        ),
        "writing_techniques_v2": SlaDefinition(
            name="技法库新鲜度",
            target_freshness_minutes=60,
            warning_threshold_minutes=120,
            critical_threshold_minutes=240
        ),
        "novel_settings_v2": SlaDefinition(
            name="设定库新鲜度",
            target_freshness_minutes=60,
            warning_threshold_minutes=120,
            critical_threshold_minutes=240
        )
    }
    
    def __init__(self, qdrant_url: str = "http://localhost:6333"):
        self.qdrant_url = qdrant_url
        self._slas = self.DEFAULT_SLAS.copy()
        self._alerts: List[Dict] = []
    
    def check_collection_freshness(self, collection_name: str) -> Dict:
        """检查Collection新鲜度"""
        import urllib.request
        
        try:
            with urllib.request.urlopen(
                f"{self.qdrant_url}/collections/{collection_name}", 
                timeout=5
            ) as response:
                data = json.loads(response.read())
                
                # 提取最后更新时间（如果有）
                # Qdrant本身不直接提供，需要在payload中存储
                
                return {
                    "collection": collection_name,
                    "status": "reachable",
                    "checked_at": datetime.now().isoformat()
                }
        except Exception as e:
            return {
                "collection": collection_name,
                "status": "error",
                "error": str(e),
                "checked_at": datetime.now().isoformat()
            }
    
    def check_sla_compliance(self, collection_name: str, 
                             last_update_time: Optional[datetime]) -> Dict:
        """检查SLA合规性"""
        if collection_name not in self._slas:
            return {"status": "unknown", "message": "No SLA defined"}
        
        sla = self._slas[collection_name]
        
        if last_update_time is None:
            return {
                "status": "critical",
                "message": "No update time recorded",
                "sla_name": sla.name
            }
        
        age_minutes = (datetime.now() - last_update_time).total_seconds() / 60
        
        if age_minutes > sla.critical_threshold_minutes:
            status = "critical"
        elif age_minutes > sla.warning_threshold_minutes:
            status = "warning"
        elif age_minutes > sla.target_freshness_minutes:
            status = "degraded"
        else:
            status = "healthy"
        
        return {
            "sla_name": sla.name,
            "status": status,
            "age_minutes": round(age_minutes, 1),
            "target_minutes": sla.target_freshness_minutes,
            "message": f"Data is {age_minutes:.0f} minutes old (target: {sla.target_freshness_minutes} min)"
        }
    
    def generate_alert(self, severity: str, message: str, 
                     affected_collections: List[str]) -> Dict:
        """生成告警"""
        alert = {
            "id": hashlib.md5(f"{message}{datetime.now().isoformat()}".encode()).hexdigest()[:8],
            "severity": severity,  # critical, warning, info
            "message": message,
            "affected_collections": affected_collections,
            "timestamp": datetime.now().isoformat(),
            "acknowledged": False
        }
        
        self._alerts.append(alert)
        
        # 这里可以集成通知系统（邮件、Slack、PagerDuty等）
        self._send_notification(alert)
        
        return alert
    
    def _send_notification(self, alert: Dict) -> None:
        """发送告警通知"""
        # 实际实现可对接各种通知渠道
        print(f"[ALERT] [{alert['severity'].upper()}] {alert['message']}")
    
    def get_overall_health(self, pipeline_healths: List[PipelineHealth]) -> Dict:
        """获取整体健康状态"""
        if not pipeline_healths:
            return {"status": "unknown", "message": "No health data"}
        
        # 根据各管道状态综合判断
        statuses = [p.status for p in pipeline_healths]
        
        if any(s == HealthStatus.DOWN for s in statuses):
            overall = "down"
        elif any(s == HealthStatus.DEGRADED for s in statuses):
            overall = "degraded"
        else:
            overall = "healthy"
        
        # 计算SLA达成率
        total_runs = sum(p.records_processed for p in pipeline_healths)
        total_failed = sum(p.failed_records for p in pipeline_healths)
        sla_rate = (total_runs - total_failed) / total_runs if total_runs > 0 else 0
        
        return {
            "overall_status": overall,
            "sla_compliance_rate": f"{sla_rate:.2%}",
            "active_pipelines": len(pipeline_healths),
            "active_alerts": len([a for a in self._alerts if not a["acknowledged"]]),
            "checked_at": datetime.now().isoformat()
        }
    
    def generate_dashboard_report(self) -> Dict:
        """生成仪表板报告"""
        report = {
            "generated_at": datetime.now().isoformat(),
            "overall_health": self.get_overall_health([]),
            "sla_status": {},
            "active_alerts": [a for a in self._alerts if not a["acknowledged"]],
            "collection_status": {}
        }
        
        # 检查各Collection状态
        for collection in self._slas.keys():
            freshness = self.check_collection_freshness(collection)
            report["collection_status"][collection] = freshness
            
            # SLA检查
            sla_check = self.check_sla_compliance(collection, None)
            report["sla_status"][collection] = sla_check
            
            # 生成告警
            if sla_check["status"] in ["critical", "warning"]:
                self.generate_alert(
                    severity=sla_check["status"],
                    message=sla_check["message"],
                    affected_collections=[collection]
                )
        
        return report
```

## 六、总结

本次审核识别了众生界项目数据管道的以下核心问题：

1. **数据质量层**：健康检查误报、双套配置系统、质量阈值固定缺乏自适应
2. **处理效率层**：模型无并发、Qdrant连接无池化、缺乏重试熔断
3. **可观测性层**：检索质量无监控、无新鲜度告警、缺乏指标体系
4. **代码质量层**：测试覆盖不足、历史脚本堆积

建议的实施优先级：
- **P0（立即）**：修复健康检查、消除裸except、统一配置加载器
- **P1（近期）**：Qdrant连接池、检索监控、历史脚本归档
- **P2（中期）**：铜银金架构、模型服务化、自适应阈值

本方案遵循数据工程最佳实践，强调：
- **可靠性**：幂等设计、断点续传、熔断保护
- **可观测性**：端到端监控、指标驱动
- **可扩展性**：连接池化、模型服务化解耦
