### 2.3 当前问题

## 三、优化方案设计

### 方案A: 知识图谱增强检索

**核心思路**: 建立实体-技法-案例的三层关联图谱

```
┌─────────────────────────────────────────────────────────┐
│                    知识图谱增强层                         │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  实体层 (knowledge_graph)                               │
│  ├── 势力 ──┐                                          │
│  ├── 角色 ──┼──► 技法层 (techniques)                   │
│  └── 力量 ──┘       │                                  │
│                     ├── 战斗技法 ──┐                    │
│                     ├── 人物技法 ──┼──► 案例层 (cases) │
│                     └── 氛围技法 ──┘                    │
│                              │                          │
│                              ▼                          │
│                    战斗场景案例                          │
│                    ├── 案例1 (引用技法A, 涉及角色X)     │
│                    └── 案例2 (引用技法B, 涉及势力Y)     │
└─────────────────────────────────────────────────────────┘

**实现步骤**:
1. 为每个案例标注：使用的技法、涉及的角色/势力
2. 为每个技法标注：适用的实体类型
3. 建立三元组关联：`案例 --使用--> 技法 --适用--> 实体`

### 方案C: 多级检索架构

**核心思路**: 粗筛 → 细筛 → 精排

```
┌─────────────────────────────────────────────────────────┐
│                    多级检索流程                          │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  Level 1: 粗筛 (Dense向量)                             │
│  ├── 召回: 1000条                                      │
│  └── 目标: 快速缩小范围                                 │
│                                                         │
│  Level 2: 细筛 (Sparse + 过滤)                         │
│  ├── 过滤: 场景类型、题材、质量分数                     │
│  ├── 召回: 100条                                       │
│  └── 目标: 精确匹配关键词                               │
│                                                         │
│  Level 3: 精排 (重排序模型)                             │
│  ├── 重排: BGE-Reranker 或 Cross-Encoder              │
│  ├── 召回: 10条                                        │
│  └── 目标: 相关性最终排序                               │
│                                                         │
│  Level 4: 后处理 (上下文整合)                           │
│  ├── 关联: 相关技法 + 涉及设定                         │
│  └── 输出: 结构化结果                                   │
└─────────────────────────────────────────────────────────┘
```

class CreationAssistant:
    """创作智能助手"""
    
    def recommend_for_scene(self, scene_type: str, context: dict):
        """为特定场景推荐素材"""
        
        recommendations = {
            "techniques": [],  # 推荐技法
            "cases": [],       # 推荐案例
            "settings": [],    # 推荐设定
            "warnings": []     # 注意事项
        }
        
        # 1. 根据场景类型推荐技法
        techniques = self.search_techniques(
            query=f"{scene_type} 写作技巧",
            top_k=5
        )
        
        # 2. 根据上下文角色推荐相关设定
        for char in context.get("characters", []):
            settings = self.get_character_setting(char)
            recommendations["settings"].append(settings)
        
        # 3. 检索优秀案例
        cases = self.search_cases(
            scene_type=scene_type,
            min_quality=7.0,
            top_k=3
        )
        
        # 4. 检查潜在问题
        warnings = self.check_consistency(context)
        
        return recommendations

### B. 多级检索实现

```python
# modules/knowledge_base/multi_level_search.py

class MultiLevelSearch:
    """多级检索管理器"""
    
    def search_cases(
        self,
        query: str,
        scene_type: str = None,
        min_quality: float = 0.0,
        top_k: int = 10
    ) -> list:
        """多级检索案例"""
        
        # Level 1: Dense粗筛
        dense_results = self.dense_search(query, limit=1000)
        
        # Level 2: Sparse + 过滤细筛
        filtered = []
        for r in dense_results:
            if scene_type and r.payload.get("scene_type") != scene_type:
                continue
            if r.payload.get("quality_score", 0) < min_quality:
                continue
            filtered.append(r)
        
        filtered = filtered[:100]
        
        # Level 3: Reranker精排
        reranked = self.rerank(query, filtered)
        
        return reranked[:top_k]
    
    def rerank(self, query: str, candidates: list) -> list:
        """使用Reranker重排序"""
        
        from sentence_transformers import CrossEncoder
        
        model = CrossEncoder('BAAI/bge-reranker-v2-m3')
        
        pairs = [(query, c.payload.get("content", "")) for c in candidates]
        scores = model.predict(pairs)
        
        # 按分数重排
        reranked = sorted(
            zip(candidates, scores),
            key=lambda x: x[1],
            reverse=True
        )
        
        return [r[0] for r in reranked]
```

### 8.3 提炼实现脚本

```python
# tools/extract_techniques_from_novels.py

"""
从原始小说自动提炼创作技法

使用方法:
    # 提炼单本小说
    python extract_techniques_from_novels.py --novel "小说名.txt"
    
    # 批量提炼
    python extract_techniques_from_novels.py --batch
    
    # 按场景类型提炼
    python extract_techniques_from_novels.py --scene-type "战斗"
"""

import json
import re
from pathlib import Path
from typing import List, Dict
from openai import OpenAI

# 路径配置
NOVEL_DIR = Path("E:/小说资源")
TECHNIQUE_DIR = Path("D:/动画/众生界/创作技法")
OUTPUT_DIR = Path("D:/动画/众生界/.extracted_techniques")

# 技法维度映射
DIMENSIONS = {
    "战斗": "04-战斗冲突维度",
    "人物": "03-人物维度",
    "剧情": "02-剧情维度",
    "氛围": "05-氛围意境维度",
    "叙事": "06-叙事维度",
    "情感": "08-情感维度",
    "世界观": "01-世界观维度",
}

# LLM 提示词
TECHNIQUE_EXTRACTION_PROMPT = """
你是一位资深小说写作技法分析师。请分析以下小说片段，提炼出其中使用的创作技法。

## 小说片段
{content}

## 分析要求

请识别并提取以下类型的技法：

1. **战斗技法**：战斗节奏、代价设计、战术逻辑、群体牺牲描写
2. **人物技法**：心理描写、成长弧光、群像塑造、道德灰色
3. **剧情技法**：伏笔设计、悬念布局、反转策划、冲突结构
4. **氛围技法**：意境营造、克制表达、痕迹式描写、对比结构
5. **叙事技法**：POV切换、时间折叠、节奏控制
6. **情感技法**：集体情感、历史创伤、动作暗示

## 输出格式

```json
{{
  "techniques": [
    {{
      "name": "技法名称",
      "dimension": "所属维度",
      "description": "技法简述",
      "example": "原文中的典型例子",
      "application": "适用场景"
    }}
  ]
}}
```

注意：
- 每个技法需要具体、可操作
- 标注技法所属的11维度之一
- 引用原文作为示例
- 说明技法的适用场景
"""


class TechniqueExtractor:
    """技法提炼器"""
    
    def __init__(self):
        self.client = OpenAI()
        self.existing_techniques = self._load_existing_techniques()
    
    def _load_existing_techniques(self) -> List[str]:
        """加载现有技法名称列表"""
        techniques = []
        for md_file in TECHNIQUE_DIR.rglob("*.md"):
            content = open(md_file, 'r', encoding='utf-8').read()
            # 提取技法名称
            matches = re.findall(r'^## [一二三四五六七八九十]+、[^：]*：?(.+)$', content, re.M)
            techniques.extend([m.strip() for m in matches])
        return techniques
    
    def extract_from_scene(self, content: str, scene_type: str) -> List[Dict]:
        """从场景中提炼技法"""
        
        prompt = TECHNIQUE_EXTRACTION_PROMPT.format(content=content[:3000])
        
        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            response_format={"type": "json_object"}
        )
        
        result = json.loads(response.choices[0].message.content)
        return result.get("techniques", [])
    
    def deduplicate(self, new_techniques: List[Dict]) -> List[Dict]:
        """去重，只保留新技法"""
        unique = []
        for tech in new_techniques:
            name = tech.get("name", "")
            if name and name not in self.existing_techniques:
                unique.append(tech)
        return unique
    
    def save_technique(self, technique: Dict, source_novel: str):
        """保存技法到对应维度目录"""
        dimension = technique.get("dimension", "元维度")
        
        # 确定保存目录
        dim_map = {
            "战斗冲突维度": "04-战斗冲突维度",
            "人物维度": "03-人物维度",
            "剧情维度": "02-剧情维度",
            "氛围意境维度": "05-氛围意境维度",
            "叙事维度": "06-叙事维度",
            "情感维度": "08-情感维度",
            "世界观维度": "01-世界观维度",
        }
        
        target_dir = TECHNIQUE_DIR / dim_map.get(dimension, "10-元维度")
        target_dir.mkdir(parents=True, exist_ok=True)
        
        # 生成技法文件
        content = f"""# {technique['name']}

> **来源小说**: {source_novel}
> **所属维度**: {dimension}

---

## 技法简述

{technique['description']}

---

## 原文示例

{technique['example']}

---

## 适用场景

{technique['application']}
"""
        
        # 保存
        safe_name = re.sub(r'[\\/*?:"<>|]', '', technique['name'])
        output_file = target_dir / f"{safe_name}.md"
        output_file.write_text(content, encoding='utf-8')
        
        return output_file


def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="从小说提炼技法")
    parser.add_argument("--novel", type=str, help="单本小说文件名")
    parser.add_argument("--batch", action="store_true", help="批量提炼")
    parser.add_argument("--scene-type", type=str, help="按场景类型提炼")
    
    args = parser.parse_args()
    
    extractor = TechniqueExtractor()
    
    if args.novel:
        # 单本提炼
        novel_file = NOVEL_DIR / args.novel
        content = open(novel_file, 'r', encoding='utf-8', errors='ignore').read()
        techniques = extractor.extract_from_scene(content, "综合")
        
        new_techniques = extractor.deduplicate(techniques)
        print(f"发现 {len(new_techniques)} 条新技法")
        
        for tech in new_techniques:
            extractor.save_technique(tech, args.novel)


if __name__ == "__main__":
    main()
```
